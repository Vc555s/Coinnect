User Registration

URL: /register

Method: POST

Description: Register a new user with skills

Request Body:

{
    "name": "User Name",
    "email": "user@example.com",
    "skills": [
        {
            "name": "Python Programming",
            "is_offered": true,
            "availability": "weekends"
        },
        {
            "name": "Web Design",
            "is_offered": false
        }
    ]
}

Response:

{
    "message": "User registered successfully!",
    "user_id": 1
}

Get Users

URL: /users

Method: GET

Description: Get all registered users

Response:

[
    {
        "id": 1,
        "name": "Alice",
        "email": "alice@example.com",
        "trust_score": 5.0,
        "skillcoins_balance": 10.0,
        "skills": [
            {
                "skill_name": "Python Programming",
                "is_offered": true,
                "availability": "anytime"
            },
            {
                "skill_name": "Web Design",
                "is_offered": false
            }
        ]
    }
]

Match Skills

URL: /match_skills?skill_name=Python

Method: GET

Description: Match users offering a specific skill

Response:

{
    "requested_skill": "Python",
    "matches": [
        {
            "user_id": 1,
            "name": "Alice",
            "trust_score": 5.0,
            "availability": "anytime"
        }
    ]
}

Create Transaction

URL: /create_transaction

Method: POST

Description: Create a new skill transaction

Request Body:

{
    "offerer_id": 1,
    "requester_id": 2,
    "skill_id": 1,
    "amount_paid": 5.0
}

Response:

{
    "message": "Transaction successful",
    "transaction_id": 1
}
